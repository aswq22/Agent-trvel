State 数据模型（跨 Agent 共享）
class TripParams(BaseModel): # ParserAgent 输出 destination: str start_date: str days: int budget: float preferences: List[str] # ["历史文化", "美食", ...] class TravelPlanState(TypedDict): # 主图共享 State user_input: str trip_params: TripParams attractions: List[dict] # AttractionAgent 输出 route: dict # RouteAgent 输出 hotels: List[dict] # HotelAgent 输出 foods: List[dict] # FoodAgent 输出 final_plan: str # StrategyAgent 输出（流式） messages: Annotated[List, add_messages]
